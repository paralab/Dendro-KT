11 /*
12  * The recursive structure that is being mimicked:
13  *
14  * Traverse(subtree, parentData, nodeCoords[], input[], output[])
15  * {
16  * // Vectors for children to read/write.
17  * stage_nodeCoords[NumChildren][];
18  * stage_input[NumChildren][];
19  * stage_output[NumChildren][];
20  * childSummaries[NumChildren];
21  *
22  * parent2Child(parentData, subtree, nodeCoords, input, output);
23  *
24  * UserPreAction(subtree, nodeCoords, input, output);
25  *
26  * if (needToDescendFurther)
27  * {
28  * topDownNodes(subtree, nodeCoords, input, output,
29  * stage_nodeCoords, stage_input, stage_output,
30  * childSummaries);
31  *
32  * for (child_sfc = 0; child_sfc < NumChildren; child_sfc++)
33  * {
34  * Traverse(subtree.getChildSFC(child_sfc),
35  * currentData,
36  * stage_nodeCoords[child_sfc],
37  * stage_input[child_sfc],
38  * stage_output[child_sfc]);
39  * }
40  *
41  * bottomUpNodes(subtree, nodeCoords, input, output,
42  * stage_nodeCoords, stage_input, stage_output,
43  * childSummaries);
44  * }
45  *
46  * UserPostAction(subtree, nodeCoords, input, output);
47  *
48  * child2Parent(parentData, subtree, nodeCoords, input, output);
49  * }
50  *
51  *
52  * In the iterative version, the above function will be turned inside out.
53  *
54  * - The call stack will be encapsulated inside a stateful iterator.
55  *
56  * - The callbacks to UserPreAction() and UserPostAction() will be replaced
57  * by two pairs of entry-exit points whereupon program control is
58  * surrendered and regained at every level.
59  *
60  * - The user can decide whether to descend or skip descending, at any level,
61  * by calling step() or next(), respectively.
62  */

270  topDownNodes(parentFrame, &parentFrame.m_extantChildren); // Free to resize children buffers.
271 
272  parentFrame.m_numExtantChildren = 0;
273 
274  // Push incident child frames in reverse order.
275  for (ChildI child_sfc_rev = 0; child_sfc_rev < NumChildren; child_sfc_rev++)
276  {
277  const sfc::SubIndex child_sfc(NumChildren-1 - child_sfc_rev);
278  const SFC_State<dim> sfc = parentFrame.m_sfc;
279  const sfc::ChildNum child_m = sfc.child_num(child_sfc);
280 
281  if (parentFrame.m_extantChildren & (1u << child_m))
282  {
283  assert(m_stack.size() < m_stack.capacity()); // Otherwise, violated constructor max_depth.
284 
285  // Frame constructor must select a sub-list of the tree, and do treeSplitters.
286 
287  m_stack.emplace_back(
288  &parentFrame,
289  child_sfc,
290  parentFrame.m_currentSubtree.getChildMorton(child_m),
291  sfc.child_curve(child_m));
292  parentFrame.m_numExtantChildren++;
293  }
294  }
295

297  // Enter the new top frame, which represents the 0th child.
298  parent2Child(*m_stack.back().m_parentFrame, m_stack.back());
299  else
