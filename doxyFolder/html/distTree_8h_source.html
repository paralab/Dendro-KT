182  // Default domainDecider (physical)
183  static ::ibm::Partition defaultDomainDecider(const double * physCoords, double physSize)
184  {
185  // Default: keep the unit cube.
186 
187  // In = discard, Out = retain
188 
189  // In_1d: (coord + sz) <= 0.0 or coord >= 1.0 // Closed "In" contains closure of octant.
190  // Out_1d: coord > 0.0 and (coord + sz) < 1.0 // Open "Out" contains closure of octant.
191 
192  // In this case of the box,
193  // In_Nd: OR_{over dims} { In_1d() }
194  // Out_Nd: AND_{over dims} { Out_1d() }
195 
196  bool isIn = false;
197  bool isOut = true;
198 
199  for (int d = 0; d < dim; ++d)
200  {
201  isIn |= (physCoords[d] + physSize <= 0.0 or physCoords[d] >= 1.0);
202  isOut &= (physCoords[d] > 0.0 and physCoords[d] + physSize < 1.0);
203  }
204 
205  // Can be isIn, or isOut, or neither, but NEVER both.
206 
207  if (isIn && !isOut)
208  return ::ibm::IN;
209  else if (isOut && !isIn)
210  return ::ibm::OUT;
211  else if (!isIn && !isOut)
212  return ::ibm::INTERCEPTED;
213  else
214  throw std::logic_error("Filter function returned both isIn and isOut.");
215  }
