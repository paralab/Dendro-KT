65  : MeshLoopImpl<T, dim, TreatSorted>(tnlist, sz, visitEmpty, visitPre, visitPost)

251 template <typename T, unsigned int dim, bool TreatSorted>
252 bool MeshLoopImpl<T, dim, TreatSorted>::step(bool vEmpty, bool vPre, bool vPost)
253 {
254  if (m_stack.size() == 0)
255  throw std::out_of_range("Exited from root subtree. No more elements");
256  if (!isPre())
257  return next(vEmpty, vPre, vPost);
258 
259  m_stack.reserve(m_stack.size() + NumChildren);
260 
261  MeshLoopFrame<T, dim> &parentFrame = getTop();
262  parentFrame.m_is_pre = false;
263 
264  if (parentFrame.m_lev < m_uiMaxDepth)
265  for (ChildI child_sfc_rev = 0; child_sfc_rev < NumChildren; ++child_sfc_rev)
266  {
267  // Figure out child_m and cRot.
268  const ChildI child_sfc = NumChildren - 1 - child_sfc_rev;
269  const RotI pRot = parentFrame.m_pRot;
270  const ChildI * const rot_perm = &rotations[pRot*2 * NumChildren + 0 * NumChildren];
271  const ChildI child_m = rot_perm[child_sfc];
272  const ChildI cRot = HILBERT_TABLE[pRot * NumChildren + child_m];
273 
274  RankI ch_begin = parentFrame.m_child_splitters[child_sfc];
275  RankI ch_end = parentFrame.m_child_splitters[child_sfc+1];
276  LevI ch_lev = parentFrame.m_lev + 1;
277 
278  if ((vPre || vPost) && (ch_end > ch_begin || vEmpty))
279  {
280  bucketAndPush(ch_begin, ch_end, ch_lev, cRot);
281  }
282  }
283 
284  return isPre();
285 }
