158  unsigned int lev,
159  MPI_Comm comm)
160 {

166

168

170  // as well as size of our partition.
171  const RankI totalCount = 1u << (dim * lev);
172  const RankI myCount = (totalCount / nProc) + (rProc < totalCount % nProc);
173  const RankI prevCount = (totalCount / nProc) * rProc
174  + (rProc < totalCount % nProc ? rProc : totalCount % nProc);
175 
176  //
177  // Generate points in the Morton order before sorting in SFC order.
178  //
179  if (myCount)
180  {
181  // Breadth-first tree traversal, compute ranks to determine acceptance.
182  // Invariant: The TreeNodes in out are exactly the subtrees of the current
183  // level that have a nonempty overlap with designated rank interval.
184  //
185  std::vector<TreeNode<T, dim>> tnBuffer;
186  std::vector<size_t> rankBuffer, rankOut;
187 
188  out.emplace_back(); // Root of tree. Definitely overlaps.
189  rankOut.emplace_back(0); //
190 
191  while (/*tnBuffer.size() > 0 && */ out[0].getLevel() < lev)
192  {
193  for (size_t ii = 0; ii < out.size(); ii++)
194  {
195  const TreeNode<T, dim> &parent = out[ii];
196  const size_t &parentRank = rankOut[ii];
197  const size_t childSize = 1u << (dim * (lev - parent.getLevel() - 1));
198 
199  // Test each child for intersection with rank interval.
200  for (ChildI child_m = 0; child_m < (1u << dim); child_m++)
201  {
202  const size_t childRank = parentRank + childSize * child_m;
203  if (prevCount < childRank + childSize &&
204  childRank < prevCount + myCount)
205  {
206  tnBuffer.emplace_back(parent.getChildMorton(child_m));
207  rankBuffer.emplace_back(childRank);
208  }
209  }
210 
211  }
212 
213  // Swap buffers.
214  out.clear();
215  rankOut.clear();
216  std::swap(tnBuffer, out);
217  std::swap(rankBuffer, rankOut);
218  }
219  }
220 
221  //
222  // Re-partition the points according to the SFC order.
223  //
224  SFC_Tree<T, dim>::distTreeSort(out, 0.0, comm);
225 
226  return 1;

828 template <typename T, unsigned int dim>
829 void checkSiblingLeafsTogether(std::vector<TreeNode<T, dim>> &tree,
830  MPI_Comm nonemptys,
831  bool &isReceiver,
832  bool &isSender,
833  RankI &srcRankFirst,
834  RankI &srcRankLast,
835  RankI &destRank,
836  unsigned int &countFront,
837  unsigned int &countBack)
838 {
839  int nNE, rNE;
840  MPI_Comm_rank(nonemptys, &rNE);
841  MPI_Comm_size(nonemptys, &nNE);
842 
843  // Algorithm idea:
844  //
845  // - Goal: Move separated sibling leafs onto the lowest rank that contains
846  // one of the siblings.
847  // - Each rank needs to know if it is sending and/or receiving, and if so,
848  // what is the destination and/or source(s).
849  // - To figure this out, each rank sends a query to its left (resp. right),
850  // asking for the least (resp. greatest) rank containing a sibling of the
851  // front (resp. back) of the local tree partition, as well as count
852  // of the number of siblings. (Count, b/c we don't transfer if not all
853  // the siblings are accounted for, e.g. because children of a sibling
854  // are present instead.)
855  // - The leftward and rightward ranks are responsible to answer the
856  // corresponding query.
857  // - Usually the answer to the query can be determined and sent immediately.
858  // - However, if all TreeNodes on a rank have the same parent, the answer
859  // to the given query depends on receiving an answer from the next rank.
860  // Small sequential chains of dependencies will be created in this case.
861  // The upper bound to the size of such a chain is NUM_CHILDREN.
862 
863 
864  // Answer[0]: endpoint rank. Answer[1]: inclusive sibling count.
865  //
866  struct Answer { RankI a[2]; };
867  Answer myLeftAnswer, myRightAnswer;
868  Answer externRightAnswer, externLeftAnswer;
869 
870  TreeNode<T, dim> externRightQuery, externLeftQuery;
871  TreeNode<T, dim> myLeftQuery, myRightQuery;
872 
873  myLeftQuery = tree.front().getParent();
874  myRightQuery = tree.back().getParent();
875 
876  MPI_Request requestMLQ, requestMRQ, requestERA, requestELA;
877  MPI_Status status;
878 
879  // Ask queries.
880  if (rNE > 0)
881  par::Mpi_Isend<TreeNode<T, dim>>(&myLeftQuery, 1, rNE-1, 0, nonemptys, &requestMLQ);
882  if (rNE < nNE - 1)
883  par::Mpi_Isend<TreeNode<T, dim>>(&myRightQuery, 1, rNE+1, 0, nonemptys, &requestMRQ);
884 
885  // Counts are used in/with the answers.
886  countFront = 0;
887  countBack = 0;
888  while (countFront < tree.size() && tree[countFront].getParent() == myLeftQuery)
889  countFront++;
890  while (countBack < tree.size() && tree[tree.size()-1 - countBack].getParent() == myRightQuery)
891  countBack++;
892 
893  // If the rank is an endpoint of the comm, already have an answer on that end.
894  if (rNE == 0)
895  myLeftAnswer = {{rNE, countFront}};
896  if (rNE == nNE - 1)
897  myRightAnswer = {{rNE, countBack}};
898 
899  // Listen for query from left neighbour, and answer it.
900  if (rNE > 0)
901  {
902  par::Mpi_Recv<TreeNode<T, dim>>(&externLeftQuery, 1, rNE-1, 0, nonemptys, &status);
903 
904  // Cases we can answer immediately. Postpone receiving our own answer.
905  if (!(externLeftQuery == tree.back().getParent()))
906  {
907  unsigned int count = 0;
908  while (count < tree.size() && tree[count].getParent() == externLeftQuery)
909  count++;
910  externLeftAnswer = (count > 0 ? Answer{{rNE, count}} : Answer{{rNE-1, 0}});
911  par::Mpi_Isend<RankI>(externLeftAnswer.a, 2, rNE-1, 66, nonemptys, &requestELA);
912  }
913  else if (rNE == nNE - 1)
914  par::Mpi_Isend<RankI>(myRightAnswer.a, 2, rNE-1, 66, nonemptys, &requestELA);
915 
916  // In this case must receive our own answer before giving an answer.
917  else
918  {
919  // Create dependency.
920  par::Mpi_Recv<RankI>(myRightAnswer.a, 2, rNE+1, 66, nonemptys, &status);
921  myRightAnswer.a[1] += countBack;
922  par::Mpi_Isend<RankI>(myRightAnswer.a, 2, rNE-1, 66, nonemptys, &requestELA);
923  }
924  }
925 
926  // Listen for query from right neighbour, and answer it.
927  if (rNE < nNE - 1)
928  {
929  par::Mpi_Recv<TreeNode<T, dim>>(&externRightQuery, 1, rNE+1, 0, nonemptys, &status);
930 
931  // Cases we can answer immediately. Postpone receiving our own answer.
932  if (!(externRightQuery == tree.front().getParent()))
933  {
934  unsigned int count = 0;
935  while (count < tree.size() && tree[tree.size()-1 - count].getParent() == externRightQuery)
936  count++;
937  externRightAnswer = (count > 0 ? Answer{{rNE, count}} : Answer{{rNE+1, 0}});
938  par::Mpi_Isend<RankI>(externRightAnswer.a, 2, rNE+1, 66, nonemptys, &requestERA);
939  }
940  else if (rNE == 0)
941  par::Mpi_Isend<RankI>(myLeftAnswer.a, 2, rNE+1, 66, nonemptys, &requestERA);
942 
943  // In this case must receive our own answer before giving an answer.
944  else
945  {
946  // Create dependency.
947  par::Mpi_Recv<RankI>(myLeftAnswer.a, 2, rNE-1, 66, nonemptys, &status);
948  myLeftAnswer.a[1] += countFront;
949  par::Mpi_Isend<RankI>(myLeftAnswer.a, 2, rNE+1, 66, nonemptys, &requestERA);
950  }
951  }
952 
953  // Revisit cases for which we haven't yet received our answer.
954  if (rNE < nNE - 1 && (rNE == 0 || !(externLeftQuery == tree.back().getParent())))
955  {
956  par::Mpi_Recv<RankI>(myRightAnswer.a, 2, rNE+1, 66, nonemptys, &status);
957  myRightAnswer.a[1] += countBack;
958  }
959  if (rNE > 0 && (rNE == nNE - 1 || !(externRightQuery == tree.front().getParent())))
960  {
961  par::Mpi_Recv<RankI>(myLeftAnswer.a, 2, rNE-1, 66, nonemptys, &status);
962  myLeftAnswer.a[1] += countFront;
963  }
964 
965  // Wait for sends to finish.
966  if (rNE > 0)
967  {
968  MPI_Wait(&requestMLQ, &status);
969  MPI_Wait(&requestELA, &status);
970  }
971  if (rNE < nNE - 1)
972  {
973  MPI_Wait(&requestMRQ, &status);
974  MPI_Wait(&requestERA, &status);
975  }
976 
977 
978  // Our queries are now answered, and we have enough info
979  // to send/receive TreeNodes. Prepare this info.
980 
981  // Output: isReceiver, isSender
982  // srcRankFirst, srcRankLast, destRank
983  // countFront, countBack
984 
985  const unsigned int NUM_CHILDREN = (1u << dim);
986 
987  isReceiver = (myRightAnswer.a[1] == NUM_CHILDREN);
988  isSender = (tree.front().getParent() == tree.back().getParent()
989  ? (myLeftAnswer.a[1] + myRightAnswer.a[1] - tree.size() == NUM_CHILDREN)
990  : (myLeftAnswer.a[1] == NUM_CHILDREN));
991 
992  srcRankFirst = rNE + 1;
993  srcRankLast = myRightAnswer.a[0];
994  destRank = myLeftAnswer.a[0];
995 
996  // Make sure communication involves not just ourselves.
997  isReceiver = isReceiver && (srcRankLast >= srcRankFirst);
998  isSender = isSender && (destRank < rNE);
999 }

1003 void keepSiblingLeafsTogether(std::vector<TreeNode<T, dim>> &tree, MPI_Comm comm)
1004 {
1005  int nProc, rProc;
1006  MPI_Comm_rank(comm, &rProc);
1007  MPI_Comm_size(comm, &nProc);
1008 
1009  // If some ranks, have no TreeNodes, exclude them from the new communicator.
1010  MPI_Comm nonemptys;
1011  MPI_Comm_split(comm, (tree.size() > 0 ? 1 : MPI_UNDEFINED), rProc, &nonemptys);
1012 
1013  if (!tree.size())
1014  return;
1015 
1016  bool isSender, isReceiver;
1017  RankI srcRankFirst, srcRankLast, destRank;
1018  unsigned int countFront, countBack;
1019 
1020  MPI_Status status;
1021 
1022  int nNE, rNE;
1023  MPI_Comm_rank(nonemptys, &rNE);
1024  MPI_Comm_size(nonemptys, &nNE);
1025 
1026  checkSiblingLeafsTogether(tree,
1027  nonemptys,
1028  isReceiver,
1029  isSender,
1030  srcRankFirst,
1031  srcRankLast,
1032  destRank,
1033  countFront,
1034  countBack );
1035 
1036  // Perform communication, and modify tree vector.
1037  //
1038  MPI_Request requestSCount, requestSPayload;
1039 
1040  if (isSender)
1041  {
1042  par::Mpi_Isend<unsigned int>(&countFront, 1, destRank, 0, nonemptys, &requestSCount);
1043  par::Mpi_Isend<TreeNode<T, dim>>(&(*tree.begin()), countFront, destRank, 0, nonemptys, &requestSPayload);
1044  tree.erase(tree.begin(), tree.begin() + countFront);
1045  }
1046 
1047  if (isReceiver)
1048  {
1049  const int numSources = srcRankLast - srcRankFirst + 1;
1050  unsigned int recvTotal = 0;
1051 
1052  std::vector<unsigned int> recvCounts(numSources, 0);
1053  for (int srcIdx = 0; srcIdx < numSources; srcIdx++)
1054  {
1055  par::Mpi_Recv<unsigned int>(&recvCounts[srcIdx], 1, srcRankFirst + srcIdx, 0, nonemptys, &status);
1056  recvTotal += recvCounts[srcIdx];
1057  }
1058 
1059  std::vector<TreeNode<T, dim>> recvBuf(recvTotal);
1060  TreeNode<T,dim> * recvPtr = &(*recvBuf.begin());
1061  for (int srcIdx = 0; srcIdx < numSources; srcIdx++)
1062  {
1063  par::Mpi_Recv<TreeNode<T, dim>>(recvPtr, recvCounts[srcIdx], srcRankFirst + srcIdx, 0, nonemptys, &status);
1064  recvPtr += recvCounts[srcIdx];
1065  }
1066 
1067  tree.insert(tree.end(), recvBuf.begin(), recvBuf.end());
1068  }
1069 
1070  if (isSender)
1071  {
1072  MPI_Wait(&requestSCount, &status);
1073  MPI_Wait(&requestSPayload, &status);
1074  }
1075 }
