82 int compareMorton(const TreeNode<T, dim> &a, const TreeNode<T, dim> &b)
83 {
84  // Use the coordinate with the highest level difference (closest to root).
85  T maxDiffCoord = 0;
86  for (int d = 0; d < dim; d++)
87  {
88  T diffCoord = a.getX(d) ^ b.getX(d); // Will have 0's where equal.
89  maxDiffCoord = (diffCoord > maxDiffCoord ? diffCoord : maxDiffCoord);
90  }
91 
92  // Find the index of the highest level of difference.
93  T levelDiff = 0;
94  while (levelDiff < a.getLevel() && levelDiff < b.getLevel()
95  && !(maxDiffCoord & (1u << (m_uiMaxDepth - levelDiff))))
96  {
97  levelDiff++;
98  }
99 
100  // Use that level to compare child numbers.
101  // In case of descendantship, ancestor is strictly less than descendant.
102  unsigned int indexA = a.getMortonIndex(levelDiff);
103  unsigned int indexB = b.getMortonIndex(levelDiff);
104 
105  if (indexA == indexB)
106  {
107  if (a.getLevel() < b.getLevel())
108  return -1;
109  else if (a.getLevel() == b.getLevel())
110  return 0;
111  else
112  return 1;
113  }
114  else
115  {
116  if (indexA < indexB)
117  return -1;
118  else
