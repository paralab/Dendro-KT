802  template <typename T>
803  MinMeanMax<T> Mpi_ReduceMinMeanMax(T localData, MPI_Comm comm)
804  {
805  MinMeanMax<T> g;
806  double localDataFloat = localData;
807 
808  Mpi_Reduce(&localData, &g.m_glob_min, 1, MPI_MIN, 0, comm);
809  Mpi_Reduce(&localDataFloat, &g.m_glob_mean, 1, MPI_SUM, 0, comm);
810  Mpi_Reduce(&localData, &g.m_glob_max, 1, MPI_MAX, 0, comm);
811 
812  int npes, rank;
813  MPI_Comm_size(comm, &npes);
814  MPI_Comm_rank(comm, &rank);
815 
816  g.m_glob_mean /= double(npes);
817 
818  if (rank != 0)
819  {
820  g.m_glob_min = 0;
821  g.m_glob_mean = 0;
822  g.m_glob_max = 0;
823  }
824 
825  return g;
826  }
