942  // - If child has hanging nodes, interpolate from parent.
943  // - Note: Any interpolated nonhanging nodes will be overwritten anyway.
944  //

1702  // Bottom-up Outline:
1703  // =========================
1704  // - Read from summary (#nodes, finest node level) per child.
1705  // - Note: A child is a leaf iff finest node level == subtree level.
1706  // - Note: A child is a leaf with hanging nodes if #nodes < npe.
1707  //
1708  // - Allocate parent output nodes and initialize to 0.

1712  // - Else, find in same order as they appear in parent.
1713  // - After receiving value from child, overwrite the child value with 0.
1714  //
1715  // - For each child:
1716  // - If child has hanging nodes, interpolate-transpose in place in child buffer.
1717  // - Pass through parent nodes.
1718  // Accumulate into parent level nodes from child buffer lex position.
1719  // ========================================================================
1720

1722  const TreeNode<unsigned int,dim> & parSubtree = this->getCurrentSubtree();
1723  const NodeT zero = 0;

1726  std::array<size_t, NumChildren> childNodeOffsets;
1727  std::array<LevI, NumChildren> childFinestLevel;

1732  //
