107  template <typename...X>
108  void send(int destIdx, const X&...scalars) {
109  static_assert(sizeof...(X) == LEN, "Must send all LEN scalars in a single call to send().");
110  assert(destIdx < m_partners->nDest());
111  assert(not m_sent[destIdx]);
112 
113  ScalarT *stage = &m_sendScalar[destIdx * LEN];
114  DENDRO_FOR_PACK( *(stage++) = scalars );
115  par::Mpi_Isend(&(m_sendScalar[destIdx * LEN]), LEN, dest(destIdx), 0, comm(), &(*m_requests[destIdx]));
116  m_sent[destIdx] = true;
117  }

119  template <typename...X>
120  void recv_all(int srcIdx, X&...scalars) {
121  assert(srcIdx < m_partners->nSrc());
122 
123  MPI_Status status;
124  if (not m_rcvd[srcIdx])
125  par::Mpi_Recv(&(m_recvScalar[srcIdx * LEN]), LEN, src(srcIdx), 0, comm(), &status);
126  m_rcvd[srcIdx] = true;
127  ScalarT *stage = &(m_recvScalar[srcIdx * LEN]);
128  DENDRO_FOR_PACK( scalars = *(stage++) );
129  }
