446  // =========================
447  // Top-down Outline:
448  // =========================
449  // - First pass: Count (#nodes, finest node level) per child.
450  // - Note: A child is a leaf iff finest node level == subtree level.
451  // - Note: A child is a leaf with hanging nodes if #nodes < npe.
452  //
453  // - Allocate child input nodes (with at least npe per child).
454  //
455  // - For each child:
456  // - If child has hanging nodes, interpolate from parent.
457  // - Note: Any interpolated nonhanging nodes will be overwritten anyway.
458  //
459  // - Second pass: Duplicate parent nodes into children.
460  // - If a child is a leaf and #nonhanging nodes <= npe, copy into lex position.
461  // - Else copy nodes into same order as they appear in parent.
462  // ========================================================================
463 
464 #ifdef DENDRO_KT_MATVEC_BENCH_H
465  bench::t_topdown.start();
466 #endif

468  const unsigned npe = intPow(m_eleOrder+1, dim);
469  const TreeNode<unsigned int,dim> & parSubtree = this->getCurrentSubtree();
470 
471  std::array<size_t, NumChildren> childNodeCounts;
472  std::array<LevI, NumChildren> childFinestLevel;
473  std::array<size_t, NumChildren> childBdryCounts;
474  childNodeCounts.fill(0);
475  childFinestLevel.fill(0);
476  childBdryCounts.fill(0);

478  *extantChildren = parentFrame.getExtantTreeChildrenMorton();

481  const size_t numInputNodes = parentFrame.mySummaryHandle.m_subtreeNodeCount;

485  for (sfc::SubIndex child_sfc(0); child_sfc < nchild(dim); ++child_sfc)

551

604  // Non-hanging node values will be overwritten later, not to worry.
605  constexpr bool transposeFalse = false;

607  &(*m_parentNodeVals.begin()),

611  }
612  }
613  }

620  //

623  numInputNodes,
624  this->getCurrentRotation(),
625  *extantChildren ))
626  {
627  const ChildI child_sfc = nodeInstance.getChild_sfc();
628  const size_t nIdx = nodeInstance.getPNodeIdx();

692  // - Note: A child is a leaf with hanging nodes if #nodes < npe.
693  //
694  // - Allocate parent output nodes and initialize to 0.
695  //
696  // - Pass through parent nodes. Accumulate nonhanging values from child output.
697  // - If a child is a leaf and #nonhanging nodes <= npe, find in lex position.
698  // - Else, find in same order as they appear in parent.
699  // - After receiving value from child, overwrite the child value with 0.
700  //
701  // - For each child:
702  // - If child has hanging nodes, interpolate-transpose in place in child buffer.
703  // - Pass through parent nodes.
704  // Accumulate into parent level nodes from child buffer lex position.
705  // ========================================================================
706 
707 #ifdef DENDRO_KT_MATVEC_BENCH_H
708  bench::t_bottomup.start();
709 #endif
710 
711  const unsigned npe = intPow(m_eleOrder+1, dim);
712  const TreeNode<unsigned int,dim> & parSubtree = this->getCurrentSubtree();
713  const NodeT zero = 0;
714 
715  std::array<size_t, NumChildren> childNodeCounts;
716  std::array<size_t, NumChildren> childNodeOffsets;
717  std::array<LevI, NumChildren> childFinestLevel;
718  childNodeOffsets.fill(0);
719 
720  //
721  // Retrieve child summaries.
722  //
723  bool thereAreHangingNodes = false;
724  MatvecBaseSummary<dim> (&summaries)[NumChildren] = parentFrame.childSummaries;
725  for (ChildI child_sfc = 0; child_sfc < NumChildren; child_sfc++)
726  {
727  childFinestLevel[child_sfc] = summaries[child_sfc].m_subtreeFinestLevel;
728  childNodeCounts[child_sfc] = summaries[child_sfc].m_subtreeNodeCount;
729 
730  if (childNodeCounts[child_sfc] > 0 && childNodeCounts[child_sfc] < npe)
731  thereAreHangingNodes = true;
732  }
733 
734  const std::vector<TreeNode<unsigned int, dim>> &myNodes = parentFrame.template getMyInputHandle<0>();
736  const size_t numParentNodes = myNodes.size();
737 
738  std::vector<NodeT> &myOutNodeValues = parentFrame.template getMyOutputHandle<0>();
739  myOutNodeValues.clear();
740  myOutNodeValues.resize(m_ndofs * numParentNodes, zero);
741 
742  std::array<TreeNode<unsigned int, dim>, NumChildren> childSubtreesSFC;
743  for (sfc::SubIndex child_sfc(0); child_sfc < nchild(dim); ++child_sfc)
744  {
745  const SFC_State<dim> sfc = this->getCurrentRotation();
746  childSubtreesSFC[child_sfc] = parSubtree.getChildMorton(sfc.child_num(child_sfc));
747  }
748 
749  //
750  // Accumulate non-hanging node values from child buffers into parent frame.
751  //
752  for (const auto &nodeInstance : IterateNodesToChildren<dim>( parSubtree,
753  &(*myNodes.begin()),
754  numParentNodes,
755  this->getCurrentRotation(),
756  extantChildren ))
757  {
758  const ChildI child_sfc = nodeInstance.getChild_sfc();
759  const size_t nIdx = nodeInstance.getPNodeIdx();
760  const size_t childOffset = childNodeOffsets[child_sfc];
761 
762  auto &childOutput = parentFrame.template getChildOutput<0>(child_sfc);
763  if (childOutput.size() > 0)
764  {
765  if (childFinestLevel[child_sfc] > parSubtree.getLevel() + 1) // Nonleaf
766  {
767  // Nodal values.
768  for (int dof = 0; dof < m_ndofs; dof++)
769  myOutNodeValues[m_ndofs * nIdx + dof]
770  += childOutput[m_ndofs * childOffset + dof];
771 
772  childNodeOffsets[child_sfc]++;
773  }
774  else // Leaf
775  {
776  const unsigned int nodeRank = TNPoint<unsigned int, dim>::get_lexNodeRank(
777  childSubtreesSFC[child_sfc],
778  myNodes[nIdx],
779  m_eleOrder );
780  assert(nodeRank < npe);
781 
782  // Nodal values.
783  for (int dof = 0; dof < m_ndofs; dof++)
784  myOutNodeValues[m_ndofs * nIdx + dof]
785  += childOutput[m_ndofs * nodeRank + dof];
786 
787  // Zero out the values after they are transferred.
788  // This is necessary so that later linear transforms are not contaminated.
789  std::fill_n( &parentFrame.template getChildOutput<0>(child_sfc)[m_ndofs * nodeRank],
790  m_ndofs, zero );
791  }
792  }
793  else
794  {
795  // TODO emit warning to log
796  // Warning: Did you forget to overwriteNodeValsOut() ?
797  }
798  }
799 
800  //
801  // Perform any needed transpose-interpolations.
802  //
803  if (thereAreHangingNodes)
804  {
805  // Initialize intermediate parent lexicographic buffer.
806  std::fill(m_parentNodeVals.begin(), m_parentNodeVals.end(), zero);
807 
808  // Use transpose of interpolation operator on each hanging child.
809  for (sfc::SubIndex child_sfc(0); child_sfc < nchild(dim); ++child_sfc)
810  {
811  auto &childOutput = parentFrame.template getChildOutput<0>(child_sfc);
812  const sfc::ChildNum::Type child_m = this->getCurrentRotation().child_num(child_sfc);
813  if (childNodeCounts[child_sfc] > 0 && childNodeCounts[child_sfc] < npe
814  && childOutput.size() > 0)
815  {
816  // Has hanging nodes. Interpolation-transpose.
817  constexpr bool transposeTrue = true;
818  m_interp_matrices.template IKD_ParentChildInterpolation<transposeTrue>(
819  &(*childOutput.begin()),
820  &(*childOutput.begin()),
821  m_ndofs,
822  child_m);
823 
824  for (int nIdxDof = 0; nIdxDof < m_ndofs * npe; nIdxDof++)
825  m_parentNodeVals[nIdxDof] += parentFrame.template getChildOutput<0>(child_sfc)[nIdxDof];
826  }
827  }
828 
829  // Accumulate from intermediate parent lex buffer to parent output.
830  for (size_t nIdx = 0; nIdx < numParentNodes; nIdx++)
831  {
832  if (myNodes[nIdx].getLevel() == parSubtree.getLevel())
833  {
834  const unsigned int nodeRank =
835  TNPoint<unsigned int, dim>::get_lexNodeRank( parSubtree,
836  myNodes[nIdx],
837  m_eleOrder );
838  assert(nodeRank < npe);
839  for (int dof = 0; dof < m_ndofs; dof++)
840  myOutNodeValues[m_ndofs * nIdx + dof]
841  += m_parentNodeVals[m_ndofs * nodeRank + dof];
842  }
843  }
844  }
845 
846  // Clean slate for next iteration, and detect nothing written by overwriteNodeValsOut.
847  for (ChildI child_sfc = 0; child_sfc < NumChildren; child_sfc++)
848  parentFrame.template getChildOutput<0>(child_sfc).resize(0);
849 
850 #ifdef DENDRO_KT_MATVEC_BENCH_H
851  bench::t_bottomup.stop();
852 #endif

858  template <typename T, unsigned int dim>
859  void fillAccessNodeCoordsFlat(bool useTNCoords,
860  const std::vector<TreeNode<T, dim>> &nodeCoords,
861  const TreeNode<T, dim> &subtree,
862  unsigned int eleOrder,
863  std::vector<double> &floatCoords)
864  {
865  const unsigned int curLev = subtree.getLevel();
866 
867  const double domainScale = 1.0 / double(1u << m_uiMaxDepth);
868  const double elemSz = double(1u << m_uiMaxDepth - curLev) / double(1u << m_uiMaxDepth);
869  double translate[dim];
870  for (int d = 0; d < dim; d++)
871  translate[d] = domainScale * subtree.getX(d);
872 
873  std::array<unsigned int, dim> numerators;
874  unsigned int denominator;
875 
876  const size_t numNodes = (useTNCoords ? nodeCoords.size() : intPow(eleOrder+1, dim));
877  floatCoords.resize(dim * numNodes);
878 
879  if (useTNCoords)
880  {
881  for (size_t nIdx = 0; nIdx < numNodes; nIdx++)
882  {
883  TNPoint<unsigned int, dim>::get_relNodeCoords(
884  subtree, nodeCoords[nIdx], eleOrder,
885  numerators, denominator);
886 
887  for (int d = 0; d < dim; ++d)
888  floatCoords[nIdx * dim + d] =
889  translate[d] + elemSz * numerators[d] / denominator;
890  }
891  }
892  else
893  {
894  denominator = eleOrder;
895  numerators.fill(0);
896  for (size_t nIdx = 0; nIdx < numNodes; nIdx++)
897  {
898  for (int d = 0; d < dim; ++d)
899  floatCoords[nIdx * dim + d] =
900  translate[d] + elemSz * numerators[d] / denominator;
901  incrementBaseB<unsigned int, dim>(numerators, eleOrder+1);
902  }
903  }
904  }

920  template <unsigned int dim>
921  std::bitset<(1u << dim)> vertexBoundary(const std::vector<bool> &nodeBdry, unsigned int eleOrder)
922  {
923  constexpr size_t numVertices = 1u << dim;
924  std::bitset<numVertices> vertexBoundary; // defaults to all false.
925  for (int v = 0; v < numVertices; ++v)
926  {
927  unsigned int vertexNidx = 0;
928  unsigned int stride = 1;
929  for (int d = 0; d < dim; ++d)
930  {
931  vertexNidx += (bool(v & (1u << d)) * eleOrder) * stride;
932  stride *= (eleOrder + 1);
933  }
934  vertexBoundary[v] = nodeBdry[vertexNidx];
935  }
936 
937  return vertexBoundary;

941  // fillLeafNodeBdry()
942  template <class TreeLoopT, unsigned int dim, class FrameT>
943  void fillLeafNodeBdry(const FrameT &frame, unsigned int eleOrder, const std::vector<bool> &parentNodeBdry, std::vector<bool> &leafNodeBdry)
944  {
945  const size_t numNodes = frame.template getMyInputHandle<0>().size();
946  const TreeNode<unsigned int, dim> *nodeCoords = &(*frame.template getMyInputHandle<0>().cbegin());
947  const TreeNode<unsigned int, dim> &subtree = frame.getCurrentSubtree();
948  const unsigned int curLev = subtree.getLevel();
949 
950  leafNodeBdry.resize(0);
951  leafNodeBdry.resize(numNodes, false);
952 
953  const std::vector<bool> nodeNonhanging = TreeLoopT::getNodeNonhangingIn(frame);
954 
955  constexpr size_t numVertices = 1u << dim;
956  const std::bitset<numVertices> parentVertexBdry =
957  vertexBoundary<dim>(parentNodeBdry, eleOrder);
958 
959  std::array<char, dim> nodeIndices;
960  nodeIndices.fill(0);
961 
962  std::array<char, dim> nodeIndexOrigin;
963  for (int d = 0; d < dim; ++d)
964  nodeIndexOrigin[d] = bool(subtree.getX(d) - subtree.getParent().getX(d)) * (eleOrder);
965 
966  for (size_t nIdx = 0; nIdx < numNodes; nIdx++)
967  {
968  if (nodeNonhanging[nIdx])
969  leafNodeBdry[nIdx] = nodeCoords[nIdx].getIsOnTreeBdry();
970 
971  else
972  {
973  // Assign boundary flag (true/false) to the node.
974  // This method does not cover all cases correctly,
975  // but it does prevent faces from becoming _partially_ boundary faces.
976  std::bitset<numVertices> vertexTestBoundary;
977  vertexTestBoundary[0] = true;
978  for (int d = 0; d < dim; ++d)
979  if (0 == nodeIndices[d] + nodeIndexOrigin[d])
980  vertexTestBoundary = vertexTestBoundary;
981  else if (0 < nodeIndices[d] + nodeIndexOrigin[d] && nodeIndices[d] + nodeIndexOrigin[d] < 2*eleOrder)
982  vertexTestBoundary |= vertexTestBoundary << (1u << d);
983  else if (nodeIndices[d] + nodeIndexOrigin[d] == 2*eleOrder)
984  vertexTestBoundary = vertexTestBoundary << (1u << d);
985 
986  // Tag a node if and only if all vertices on the face are boundary nodes.
987  const bool isBoundaryNode = ((parentVertexBdry & vertexTestBoundary) == vertexTestBoundary);
988  leafNodeBdry[nIdx] = isBoundaryNode;
989  }
990 
991  incrementBaseB<char, dim>(nodeIndices, eleOrder+1);
992  }
993  }
